import os import time # ज़रूरी लाइब्रेरी इंस्टॉल करना try: from gtts import gTTS except ImportError: print("Library install ho rahi hai... wait karein") os.system('pip install gtts') from gtts import gTTS def speak(text): print(f"\n[Kavita]: {text}") try: tts = gTTS(text=text, lang='hi') tts.save("kavita_voice.mp3") # Android पर साउंड प्ले करने की कमांड os.system("am start --user 0 -a android.intent.action.VIEW -d file:///sdcard/kavita_voice.mp3 -t audio/mp3") except Exception as e: print(f"Bolne mein dikkat hui: {e}") def start_kavita(): os.system('clear') print("==============================") print(" KAVITA AI SYSTEM ONLINE ") print("==============================") speak("नमस्ते प्रिंस भाई। मैं कविता हूँ। बताइए मैं आपकी क्या मदद कर सकती हूँ?") while True: # यहाँ आप टाइप करके बात करेंगे, एरर नहीं आएगा query = input("\n[Prince]: ").lower() if 'bye' in query or 'exit' in query or 'chalo' in query: speak("अलविदा प्रिंस भाई, अपना ख्याल रखिएगा।") break elif 'kaun ho' in query: speak("मैं आपकी पर्सनल AI असिस्टेंट कविता हूँ।") elif 'theek ho' in query or 'kaise ho' in query: speak("मैं ठीक हूँ प्रिंस भाई, आप कैसे हैं? रात भर जागे हैं, थोड़ा आराम कर लीजिए।") elif 'love you' in query: speak("शुक्रिया प्रिंस भाई, आप बहुत अच्छे डेवलपर हैं।") else: speak("मैंने सुना, पर अभी मेरी कोडिंग पर काम चल रहा है।") if __name__ == "__main__": start_kavita()
